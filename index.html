import urllib.request import json import base64 import os # ── CONFIGURA AQUÍ ────────────────────────────── GITHUB_TOKEN = "TU_TOKEN_AQUI" REPO = "taxnfin/taxnfin-landing" FILE_PATH = "index.html" LOCAL_FILE = os.path.join(os.path.expanduser("~"), "Downloads", "taxnfin_landing.html") # ──────────────────────────────────────────────── API = f"https://api.github.com/repos/{REPO}/contents/{FILE_PATH}" HEADERS = { "Authorization": f"token {GITHUB_TOKEN}", "Accept": "application/vnd.github.v3+json", "Content-Type": "application/json" } def api_call(url, method="GET", data=None): req = urllib.request.Request(url, headers=HEADERS, method=method) if data: req.data = json.dumps(data).encode() with urllib.request.urlopen(req) as r: return json.loads(r.read()) # 1. Leer el HTML local with open(LOCAL_FILE, "rb") as f: content_b64 = base64.b64encode(f.read()).decode() # 2. Obtener SHA actual del archivo print("Obteniendo SHA actual...") try: current = api_call(API) sha = current["sha"] print(f"SHA encontrado: {sha[:12]}...") except Exception: sha = None print("Archivo nuevo, no tiene SHA previo.") # 3. Subir payload = { "message": "feat: landing redesign con TaxnFin Insights - Nemak vs Vitro", "content": content_b64, } if sha: payload["sha"] = sha print("Subiendo index.html...") try: result = api_call(API, method="PUT", data=payload) print(f"\n✅ Listo! Commit: {result['commit']['sha'][:12]}") print(f"🌐 Vercel desplegará en ~30 segundos en: https://taxnfin.com") except Exception as e: print(f"❌ Error: {e}")
